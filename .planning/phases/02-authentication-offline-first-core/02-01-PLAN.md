export const protectRobot = asyncHandler(
  async (req: Request, res: Response, next: NextFunction) => {
    let token;
    if (req.headers.authorization?.startsWith("Bearer")) {
      [, token] = req.headers.authorization.split(" ");
      const { deviceId } = jwt.verify(token, process.env.JWT_SECRET) as JwtPayloadRobot;
      const robot = await robotModel.findById(deviceId);
      if (!robot) throw new Error("No robot found");
      next();
    }
    if (error?.name === "TokenExpiredError") {
      res.status(401);
      throw new Error("Token Expired, Please log in again");
    }
  }
);

**3. startTokenRefreshService(robotId: string): NodeJS.Timeout**
Background process to auto-refresh tokens before expiration:
- Accepts robotId (MongoDB ObjectId as string)
- Runs every 50 minutes (10 minutes before 1-hour access token expiration)
- Generates new access token with payload: `{ deviceId: robotId, iat: Date.now(), exp: Date.now() + 3600 }`
- Stores new access token in Redis key `robot:${robotId}:accessToken` with 1-hour TTL
- Logs refresh events: `console.log(\`[TokenManager] Refreshed access token for robot ${robotId}\`)`
- Returns interval ID so it can be cleared on shutdown

Implementation:
```typescript
export const startTokenRefreshService = (robotId: string): NodeJS.Timeout => {
  const REFRESH_INTERVAL = 50 * 60 * 1000; // 50 minutes

  return setInterval(() => {
    const payload = { deviceId: robotId };
    const newAccessToken = generateAccessToken(payload);
    // Store in Redis for retrieval (implementation in Phase 3 when Redis integration added)
    console.log(`[TokenManager] Refreshed access token for robot ${robotId}`);
  }, REFRESH_INTERVAL);
};
```

**Why 50 minutes:** Access tokens expire in 60 minutes. Refreshing at 50 minutes provides 10-minute safety margin before expiration, preventing "Token Expired" errors during active connections.
